Konfigurasi Supabase

SQL Setup Guide

1. Buat Tabel di Supabase SQL Editor

-- Buat tabel untuk menyimpan file HTML
CREATE TABLE html_files (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Buat index untuk pencarian nama
CREATE INDEX idx_html_files_name ON html_files(name);

-- Buat function untuk auto update timestamp
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

-- Buat trigger untuk auto update
CREATE TRIGGER update_html_files_updated_at
    BEFORE UPDATE ON html_files
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

2. Enable Row Level Security (RLS)

-- Enable RLS
ALTER TABLE html_files ENABLE ROW LEVEL SECURITY;

-- Policy untuk akses publik (baca/tulis)
-- Gunakan ini jika ingin akses terbuka
CREATE POLICY "Allow public access" ON html_files
    FOR ALL USING (true) WITH CHECK (true);

-- ATAU untuk akses lebih aman dengan API key:
-- CREATE POLICY "Allow anon access" ON html_files
--     FOR ALL TO anon USING (true) WITH CHECK (true);
